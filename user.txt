    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        try {
            // TODO add your handling code here:
            Connection c=DriverManager.getConnection("jdbc:derby://localhost:1527/ebookshop ");
            java.sql.Statement st= c.createStatement();
            ResultSet r=st.executeQuery("select * from book");
            while(r.next())
            {
                String id=String.valueOf(r.getInt(1));
                String t=r.getString(2);
                String a=r.getString(3);
                String p=String.valueOf(r.getInt(4));
                String q=String.valueOf(r.getInt(5));
                String data[]={id,t,a,p,q};
                DefaultTableModel m=(DefaultTableModel)jTable1.getModel();
                m.addRow(data);
            }
           
        }
        catch (SQLException ex) {
            System.out.println(ex);
        }

    }                                        

    private void jComboBox1ActionPerformed(java.awt.event.ActionEvent evt) {                                           
       try
       {
            // TODO add your handling code here:
            Connection c=DriverManager.getConnection("jdbc:derby://localhost:1527/ebookshop");
            java.sql.Statement st= c.createStatement();
            ResultSet r=st.executeQuery("select author from book");
            jComboBox1.removeAllItems();
            while(r.next())
            {
               
                jComboBox1.addItem(r.getString("author"));
               
            }
           
          
            String ch=jComboBox1.getItemAt(jComboBox1.getSelectedIndex());
            ResultSet rs=st.executeQuery("select title,author,price,quantity from book where author='"+ch+"'");
            while(rs.next())
            {
                 String t=rs.getString("title");
                 String a=rs.getString("author");
                 String p=String.valueOf(rs.getInt(3));
                String q=String.valueOf(rs.getInt(4));
                 
                String data[]={t,a,p,q};
                DefaultTableModel m1=(DefaultTableModel)jTable2.getModel();
                m1.addRow(data);
            }
           
        }
        catch (SQLException ex) {
            System.out.println(ex);
        }

    }
